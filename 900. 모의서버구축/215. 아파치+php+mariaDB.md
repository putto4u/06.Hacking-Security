

## 아파치+php+마리아DB 설치

0. sudo su -
1. apt update
2. apt install -y apache2
3. apt install -y mariadb-server
4. apt install -y php php-mysql
5. apt show apache2
   
6. systemctl start apache2
7. systemctl start mariadb
8. ufw status  \  ufw enable  \ ufw status
9. ufw allow 80/tcp
10. ufw reload
11. ufw status vervose
